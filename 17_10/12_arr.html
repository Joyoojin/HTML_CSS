원래 페이지로 돌아감
<script> var city=[]; //배열 변수 선언 city[0]= "Seoul"; city[1]= "Busan"; city[2]= "Incheon"; city[3]= "Mokpo"; city[4]= "Sejong"; function printArr(){ var i; for(i=0; i<city.length; i++){ document.write("배열 데이터 [" + i + "] = "+ city[i] +"<br>") } } printArr(); /*출력결과 배열 데이터 [0] = Seoul 배열 데이터 [1] = Busan 배열 데이터 [2] = Incheon 배열 데이터 [3] = Mokpo 배열 데이터 [4] = Sejong*/ </script>
