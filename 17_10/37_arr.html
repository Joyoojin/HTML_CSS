연관 배열로 저장된 데이터 조회하기


원래 페이지로 돌아감
var data = { 'f0': 100, 'f1': 200, 'f2': 300 }; data['f3'] = 400; // 배열 데이터 저장 data.f4 = 500; // 배열 데이터 저장 document.write(data.f0 + "<br>"); // 'f0'키 데이터 조회 document.write(data.f1 + "<br>"); // 'f1'키 데이터 조회 document.write(data['f2'] + "<br>"); document.write(data['f3'] + "<br>"); document.write(data['f4'] + "<br>");
