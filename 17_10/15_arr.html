공백 데이터를 제외한 배열 연산하기


원래 페이지로 돌아감
<h3 style="color: salmon;"> 공백 데이터를 제외한 배열 연산하기 </h3> <script> var com = [95, 88, , 72, 68, , 99, 82, 78, 87]; //10명 중 8명의 점수만 입력 var getAvg; function printAvg() { var i; sum = 0; var count = 0; //입력 점수 카운트 변수 var n = com.length; document.write(n + "명의 점수 입력 </p>"); //10명의 점수 입력 for (i = 0; i < n; i++) { if (com[i] === undefined) { continue; } else { sum += com[i]; count++ } } document.write("점수를 입력한 학생 수: " + count + "명<p/>"); //점수를 입력한 학생 수: 8명 document.write("총합 : <b> " + sum + "</b><p/>"); // 총합 : 669 return (sum / count); } getAvg = printAvg(); //함수 호출 document.write("평균 : <b> " + getAvg + "</b><p/>"); // 평균 :83.625 /*출력결과 10명의 점수 입력 점수를 입력한 학생 수: 8명 총합 : 669 평균 : 83.625 */ </script>
