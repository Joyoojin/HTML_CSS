<h3 style="color: salmon;"> 같은 데이터 타입을 가진 배열 연산하기 </h3> <script> var arr=[10,20,30,40,50]; // 같은 데이터 타입요소 function printArr(){ var i, sum = 0; for(i=0; i<arr.length; i++){ sum += arr[i]; } return sum; } var result = printArr(); // 함수호출 document.write("배열 원소 합 : "+ result + "<br>") /*출력 결과 배열 원소 합 : 150 */ </script>
