다른 데이터 타입을 가진 배열 연산하기


원래 페이지로 돌아감
<h3 style="color: salmon;"> 다른 데이터 타입을 가진 배열 연산하기 </h3> <script> var arr=[10,20,30,40, '50']; //다른 데이터 타입 요소 // ----연산이 되지않고, 결과 값이 이어 붙어서 출력 function printArr(){ var i, sum =0; for(i=0; i<arr.length; i++){ sum+=arr[i]; } return sum; } var result = printArr(); document.write("배열 원소 합: " + result + "<br>"); /*출력결과 배열 원소 합: 10050 */ </script>
